# The point at which we start applying downward expansion to the signal
band_upper = -100
# The lower lever, where full expansion should be achieved. This value is subtracted from the band_upper to calculate band_lower
band_gap = 8
# Maximum amount of expansion when signal goes below the lower band
expansion = 20
# amount of slew limiting applied to the release tail of the gain detection signal
decay_ms = 100
# Once the signal dips below the lower band, the bands get shifted upwards by this amount. This reduces false triggers. 
# Once the signal passes completely above the upper band again, the level is reset.
hysteresis = 4

def slew_limit(ins):
    decay_sample = 30 / (decay_ms / 1000.0) / 48000 # 90db decay per second
    print(f"decay per sample: {decay_sample}")
    i = 1
    outs = np.zeros(len(ins))
    outs[0] = -100
    while i < len(ins):
        outs[i] = ins[i]
        # Slew limiter
        if outs[i] < outs[i-1] - decay_sample:
            outs[i] = outs[i-1] - decay_sample
        i += 1
    return outs

def hipass(ins):
    outs = np.zeros(len(ins))
    i = 1
    acc = 0
    while i < len(ins):
        acc = 0.8 * acc + 0.2 * ins[i]
        outs[i] = ins[i] - acc
        i += 1
    return outs
        
def gain_detect(ins):
    # High-pass input signal to remove any DC offset and bassy "warble"
    ins = hipass(ins)
    # Take the abs, add a very small offset as to not take log(0)
    ins = np.abs(ins) + db2gain(-150)
    i = 1
    f_val = -150
    # Convert into the Decibel domain
    db_data = gain2db(ins)
    # Apply a slew limiter to the release, controlled by the decay_ms parameter
    db_data = slew_limit(db_data)
    # simple low-pass filtering of resulting signal
    while i < len(ins):
        f_val = f_val * 0.998 + db_data[i] * 0.002
        db_data[i] = f_val
        i += 1
    return db_data

band_lower = band_upper - band_gap
band_delta = band_upper - band_lower
    
def expand(db_data):
    i = 1
    output = np.zeros(len(db_data))
    output_sigfilt = np.zeros(len(db_data))
    band_upper_series = np.zeros(len(db_data))
    band_lower_series = np.zeros(len(db_data))
    
    # bu and bl are the hysteresis-adjusted upper and lower bands
    bu = band_upper
    bl = band_lower
    
    output[0] = -100
    output_sigfilt[0] = -100
    band_upper_series[0] = bu
    band_lower_series[0] = bl
    
    while i < len(db_data):
        sample = db_data[i]
        value = 0
        
        # Apply hysteresis to bands
        if sample < bl:
            bl = band_lower + hysteresis
            bu = band_upper + hysteresis
        elif sample > bu:
            bl = band_lower
            bu = band_upper
        
        # Apply expansion based on whether we are:
        # Above the upper band (no expansion)
        if sample >= bu:
            value = sample
        # Below the lower band (full expansion)
        elif sample <= bl:
            value = sample - expansion
        # Between the bands (partial expansion based on how close we are to the upper band
        else:
            value = bu - ((sample)-(bu))/((bl)-(bu)) * (band_delta + expansion)
        
        
        # post below-threshold filtering. This applies different low-pass filtering to the signal
        # above, below and between the bands. It also filters upwards moves more heavily than downwards moves.
        if value < bl:
            if value > output[i-1]:
                output[i] = output[i-1] * 0.99998 + value * 0.000002
                output_sigfilt[i] = output_sigfilt[i-1] * 0.99998 + sample * 0.000002
            else:
                output[i] = output[i-1] * 0.9999 + value * 0.0001
                output_sigfilt[i] = output_sigfilt[i-1] * 0.9999 + sample * 0.0001
        elif value < bu:
            if value > output[i-1]:
                output[i] = output[i-1] * 0.999 + value * 0.001
                output_sigfilt[i] = output_sigfilt[i-1] * 0.999 + sample * 0.001
            else:
                output[i] = output[i-1] * 0.998 + value * 0.002
                output_sigfilt[i] = output_sigfilt[i-1] * 0.998 + sample * 0.002
        else:
            output[i] = output[i-1] * 0.8 + value * 0.2
            output_sigfilt[i] = output_sigfilt[i-1] * 0.8 + sample * 0.2
        
        band_upper_series[i] = bu
        band_lower_series[i] = bl
        i+=1
    
    # Finally, we take the difference between the output and the output_sigfilt signal, which is just a version of the
    # input signal that was filtered with the same filter coefficients as the resulting output signal. This gives a smoother results with no sudden spikes
    diff = output_sigfilt - output
    return output, diff, band_upper_series, band_lower_series, output_sigfilt
